SELECT dep.departement_nom AS nd, dep.departement_code, sous_resultat.ville_count
FROM departement AS dep
LEFT JOIN (
    SELECT v.ville_departement as ville_dep, COUNT(v.ville_departement), MIN(v.departement_nom) as ville_count
    FROM departement d
    LEFT JOIN villes_france_free v ON v.ville_departement = d.departement_code
    GROUP BY v.ville_departement
    ORDER BY ville_count DESC
) sous_resultat
ON dep.departement_code = sous_resultat.ville_dep
ORDER BY sous_resultat.ville_count DESC

FROM villes_france_free 
WHERE ville_surface > (SELECT AVG(ville_surface) FROM villes_france_free)
GROUP BY ville_nom,ville_surface
ORDER BY surf DESC
